district_ult_ecu_utopia_arcology = {
	base_buildtime = 480
	icon = district_ult_utopia_arcology
	
	zone_slots = {
		slot_city_government
		slot_city_01
		slot_city_02
	}

	show_on_uncolonized = {
		always = no
	}
	
	potential = { always = no}
	
	allow = {
	  has_modifier = ult_utopia_infrastructure_ecu
	  is_special_colony_type = no
	}
	
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 2000
		}
		upkeep = {
			energy = 5
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 6000
		job_bureaucrat_add = 200
		job_healthcare_add = 200
		planet_amenities_add = 3000
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_public_works }
		}
		modifier = {
			job_ult_product_designer_add = 200
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 300
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 300
		}
	}
}
